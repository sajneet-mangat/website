    <!-- JavaScript for the Three.js background animation -->

    <script>

        window.onload = function() {

            // --- Three.js Setup ---

            const canvas = document.getElementById('bg-canvas');

            const scene = new THREE.Scene();

            const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);

            const renderer = new THREE.WebGLRenderer({ canvas: canvas, antialias: true, alpha: true });

            

            renderer.setSize(window.innerWidth, window.innerHeight);

            renderer.setPixelRatio(window.devicePixelRatio);

            renderer.domElement.style.pointerEvents = 'none';


            // --- Particle System ---

            const particleCount = 2000;

            const positions = new Float32Array(particleCount * 3);

            const velocities = new Float32Array(particleCount * 3);

            const randoms = new Float32Array(particleCount);


            for (let i = 0; i < particleCount; i++) {

                // Initial positions of particles in a sphere shape

                const r = Math.random() * 50;

                const theta = Math.random() * Math.PI * 2;

                const phi = Math.random() * Math.PI * 2;

                positions[i * 3 + 0] = r * Math.sin(theta) * Math.cos(phi);

                positions[i * 3 + 1] = r * Math.sin(theta) * Math.sin(phi);

                positions[i * 3 + 2] = r * Math.cos(theta);


                // Initial random velocities

                velocities[i * 3 + 0] = Math.random() * 0.05 - 0.025;

                velocities[i * 3 + 1] = Math.random() * 0.05 - 0.025;

                velocities[i * 3 + 2] = Math.random() * 0.05 - 0.025;

                

                // Random value for a unique sparkle effect

                randoms[i] = Math.random();

            }


            const geometry = new THREE.BufferGeometry();

            geometry.setAttribute('position', new THREE.BufferAttribute(positions, 3));

            geometry.setAttribute('velocity', new THREE.BufferAttribute(velocities, 3));

            geometry.setAttribute('random', new THREE.BufferAttribute(randoms, 1));


            // Vertex Shader: Controls particle position and size

            const vertexShader = `

                precision highp float;

                attribute vec3 velocity;

                attribute float random;

                uniform float uTime;

                uniform vec2 uMouse;

                uniform float uAspect;

                uniform float uPullStrength;

                

                varying float vOpacity;

                

                void main() {

                    vec3 newPosition = position;

                    // Move particles based on their velocity and a time component

                    newPosition.x += velocity.x * uTime * 0.01;

                    newPosition.y += velocity.y * uTime * 0.01;

                    newPosition.z += velocity.z * uTime * 0.01;

                    

                    // Mouse interaction: Pull particles towards the cursor

                    vec2 screenPosition = (gl_Position.xy / gl_Position.w);

                    vec2 mouseToParticle = screenPosition - uMouse;

                    float dist = length(mouseToParticle);

                    if (dist < 0.5) {

                        float pullFactor = (0.5 - dist) / 0.5;

                        newPosition.xy -= mouseToParticle * pullFactor * uPullStrength;

                    }

                    

                    // Create a subtle breathing/pulsing effect

                    float pulse = sin(uTime * 0.5 + random * 10.0) * 0.5 + 0.5;

                    float size = (1.0 + pulse * 1.0) * 0.5;

                    gl_PointSize = size;

                    

                    // Fade out particles that are far away

                    vOpacity = 1.0 - (length(newPosition) / 80.0);

                    

                    gl_Position = projectionMatrix * modelViewMatrix * vec4(newPosition, 1.0);

                }

            `;


            // Fragment Shader: Controls particle color and transparency

            const fragmentShader = `

                precision highp float;

                varying float vOpacity;

                

                void main() {

                    // Create a circular shape for each particle

                    float r = 0.0;

                    vec2 coord = gl_PointCoord - vec2(0.5);

                    r = length(coord);

                    

                    float alpha = smoothstep(0.5, 0.4, r) * vOpacity;

                    

                    gl_FragColor = vec4(1.0, 1.0, 1.0, alpha); // White color with opacity

                }

            `;


            const material = new THREE.ShaderMaterial({

                uniforms: {

                    uTime: { value: 0.0 },

                    uMouse: { value: new THREE.Vector2(0, 0) },

                    uAspect: { value: window.innerWidth / window.innerHeight },

                    uPullStrength: { value: 0.15 } // Adjust how strong the mouse pull is

                },

                vertexShader: vertexShader,

                fragmentShader: fragmentShader,

                transparent: true,

                blending: THREE.AdditiveBlending,

                depthTest: false

            });


            const particles = new THREE.Points(geometry, material);

            scene.add(particles);


            camera.position.z = 5;


            // --- Event Listeners ---

            let mouse = new THREE.Vector2();

            let mouseTarget = new THREE.Vector2();


            window.addEventListener('mousemove', (e) => {

                mouseTarget.x = (e.clientX / window.innerWidth) * 2 - 1;

                mouseTarget.y = -(e.clientY / window.innerHeight) * 2 + 1;

            });

            

            // Smoothly interpolate the mouse position for a more natural feel

            function updateMouse() {

                mouse.x += (mouseTarget.x - mouse.x) * 0.1;

                mouse.y += (mouseTarget.y - mouse.y) * 0.1;

                material.uniforms.uMouse.value.set(mouse.x, mouse.y);

            }


            window.addEventListener('resize', () => {

                const width = window.innerWidth;

                const height = window.innerHeight;

                camera.aspect = width / height;

                camera.updateProjectionMatrix();

                renderer.setSize(width, height);

                material.uniforms.uAspect.value = width / height;

            });


            // --- Animation Loop ---

            const clock = new THREE.Clock();

            function animate() {

                requestAnimationFrame(animate);


                // Update uniforms

                material.uniforms.uTime.value = clock.getElapsedTime();

                updateMouse();


                renderer.render(scene, camera);

            }


            animate();

        };
